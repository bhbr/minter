update(args, redraw = true) {
	updateSelf(args)
	updateSubmobs()
	updateDependents()
	redraw()
}


updateSelf(args) {
	setAttributes(args)
}



redraw() {
	redrawSelf()
	redrawSubmobs()
	redrawDependents()
}



setAttributes(args) {
	for (let [key, value] in Object.entries(args)) {
		let attr = this[key]
		if (attr.passByValue) {
			if (attr.copyFrom != undefined) {
				attr.copyFrom(value)
			} else {
				console.error('Need to implement copyFrom!')
			}
		} else {
			attr = value
		}
	}
}

Mobjects always have a .view that is an HTML div
The mob's .anchor is the position of the top left (for now) corner in the superview

The view's width and height are the mob's .viewWidth and .viewHeight (this is bc we'd like to save "width" and "height" for e. g. Rectangles etc.)

A mob also has a transform, consisting of a rotation and scaling around the anchor (no shifting!).
